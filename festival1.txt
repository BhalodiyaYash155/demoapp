import React, { useEffect, useState } from "react";
import { ActivityIndicator, Image, SafeAreaView, StyleSheet, Text, View } from "react-native";

export default function App() {
  const [config, setConfig] = useState(null);
  const [loading, setLoading] = useState(true);

  // Fetch config from server
  useEffect(() => {
    const fetchConfig = async () => {
      try {
        const response = await fetch("demo.json"); // ðŸ”¹ Replace with your hosted JSON
        const data = await response.json();
        setConfig(data);
      } catch (error) {
        console.error("Error fetching config:", error);
        // fallback config
        setConfig({
          theme: "default",
          primaryColor: "#1976D2",
          secondaryColor: "#64B5F6",
          bannerImage: "https://via.placeholder.com/400x200.png?text=Default+Banner",
          message: "Welcome to My Shop!"
        });
      } finally {
        setLoading(false);
      }
    };

    fetchConfig();
  }, []);

  if (loading) {
    return (
      <SafeAreaView style={styles.center}>
        <ActivityIndicator size="large" color="#000" />
        <Text>Loading Theme...</Text>
      </SafeAreaView>
    );
  }

  return (
    <SafeAreaView
      style={[
        styles.container,
        { backgroundColor: config.primaryColor }
      ]}
    >
      {/* Banner */}
      <Image
        source={{ uri: config.bannerImage }}
        style={styles.banner}
        resizeMode="cover"
      />

      {/* Message */}
      <View style={styles.content}>
        <Text style={[styles.message, { color: config.secondaryColor }]}>
          {config.message}
        </Text>
      </View>
    </SafeAreaView>
  );
}

// Styles
const styles = StyleSheet.create({
  container: {
    flex: 1,
    alignItems: "center",
  },
  center: {
    flex: 1,
    justifyContent: "center",
    alignItems: "center",
  },
  banner: {
    width: "100%",
    height: 200,
  },
  content: {
    marginTop: 20,
    padding: 10,
  },
  message: {
    fontSize: 20,
    fontWeight: "bold",
    textAlign: "center",
  },
});
